 16    def add_or_update_trip(self, df_trips, new_trip, mmsi, msg_time_nanos, ship_desig, rng, lat, lon, brg, spd):
 17        '''
 18        Description: updates trip dataframe with new trip information
 19
 20        :df_trips:
 21        :new_trip:
 22        :mmsi:
 23        :msg_time_nanos:
 24        :ship_desig:
 25        :rng:
 26        :brg:
 27        :lat:
 28        :lon:
 29        :spd:
 30
 31        :returns:
 32        '''
 33        if new_trip: # start of new trip
 34            row = pd.Series({
 35                'MMSI'      : mmsi,
 36                'ZONE_ENTER': msg_time_nanos,
 37                'ZONE_EXIT' : msg_time_nanos,
 38                'DESIG'     : ship_desig,
 39                'CPA'       : rng,
 40                'CPA_TIME'  : msg_time_nanos,
 41                'MSG_INFO'  : [(msg_time_nanos, lat, lon, brg, rng, spd)]
 42                })
 43            df_trips = pd.concat([df_trips, row.to_frame().T], ignore_index=True)
 44        else: # continuation of current trip
 45            # find current trip in df_trips
 46            current_trip_idx = df_trips[df_trips['MMSI']==mmsi].index[-1]
 47            if rng < df_trips.iloc[current_trip_idx, 4]: # new cpa
 48                df_trips.iloc[current_trip_idx, 4] = rng
 49                df_trips.iloc[current_trip_idx, 5] = msg_time_nanos
 50            df_trips.iloc[current_trip_idx, 2] = msg_time_nanos
 51            msg_info = (msg_time_nanos, lat, lon, brg, rng, spd)
 52            df_trips.iloc[current_trip_idx, 6].append(msg_info)
 53
 54        return df_trips

 87        '''
 88        Given a period of time get the dataframe of trips that satisfy that time period
 89        '''
 90        
 91        return df_trips[(df_trips['ZONE_ENTER'].between(start_nanos, end_nanos)) | \
 92                        (df_trips['ZONE_EXIT'].between(start_nanos, end_nanos)) | \
 93                        ((start_nanos >= df_trips['ZONE_ENTER']) & (start_nanos <= df_trips['ZONE_EXIT'])) | \
 94                        ((end_nanos >= df_trips['ZONE_ENTER']) & (end_nanos <= df_trips['ZONE_EXIT']))]

 96    def get_daily_tracks(self, mat_h5s, mmsi_tgt):
 97        daily_tracks = {}
 98        mmsi_found = False
 99        for mat_h5 in mat_h5s:
100            try:
101                ship = mat_h5.root._f_get_child('MMSI_' + mmsi_tgt)
102                mmsi_found = True
103            except:
104                continue
105
106            # collect the relavant messages
107            for msgnum, gregtime in enumerate(ship.datenumber):
108                msg_nanos = greg_to_nanos(gregtime[0])
109                ymd = nanos_to_ymd_str(msg_nanos)
110                lat = ship.LAT[msgnum][0]
111                lon = ship.LON[msgnum][0]
112                if daily_tracks.get(ymd) is None:
113                    daily_tracks[ymd] = [(msg_nanos, lat, lon)]
114                else:
115                    daily_tracks[ymd].append((msg_nanos, lat, lon))
116
117        if not mmsi_found:
118            raise Exception("MMSI {} not found in .mat files")
119            
120        return daily_tracks

125    def _mars_trips(
126        self,
127        mat_files, 
128        df_mmsi_info, 
129        buffer_zone_max, 
130        detection_zone_max, 
131        detection_zone_min=0.0
132        ):
133
134        colnames = ['MMSI', 'ZONE_ENTER', 'ZONE_EXIT', 'DESIG', 'CPA', 'CPA_TIME', 'MSG_INFO']
135        df_trips_detect = pd.DataFrame(columns = colnames, dtype=object)
136        df_trips_buffer = pd.DataFrame(columns = colnames, dtype=object)
137        ship_zone = {}  # key = MMSI, value = current zone
138        for f in mat_files: # each .mat file corresponds to a day's worth of AIS messages
139            h5file = tables.open_file(f, mode = 'r')
140
141            for i, ship in enumerate(h5file.root): # each ship
142                if i > 0: # the first "ship" is garbage. ignore it.
143                    mmsi = int(ship.MMSI[0][0])
144                    # augment message info with additional web-scraped info
145                    ship_data = df_mmsi_info.loc[df_mmsi_info['MMSI'] == mmsi] 
146                    ship_desig = '' if ship_data.empty else ship_data["DESIG"].values[0]
147
148                    # NOTE: messages are not in time-sorted order!!! sort them.
149                    msgnum_nanos = []
150                    for msgnum, gregtime in enumerate(ship.datenumber):
151                        msgnum_nanos.append((msgnum, greg_to_nanos(gregtime[0])))
152                    msgnum_nanos = sorted(msgnum_nanos, key=lambda x: x[1])
153
154                    # iterate over the messages transmitted by a ship and formulate
155                    # "trips" in the detection and buffer zones.
156                    # NOTE: observing ~5-15 minutes between messages. TODO: more analysis
157                    for msgnum, nanos in msgnum_nanos:
158                        # ship.datenumber is an array that holds the time that 
159                        # each AIS message was recorded, and each AIS message has
160                        # info (LAT, LON, SOG, etc.) associated with it.
161                        msg_time_nanos = nanos
162                        lat = ship.LAT[msgnum][0]
163                        lon = ship.LON[msgnum][0]
164                        brg = get_bearing(lat, lon, SENSOR_LAT, SENSOR_LON)
165                        rng = get_range(lat, lon, SENSOR_LAT, SENSOR_LON)
166                        spd = int(ship.SOG[msgnum][0])
167
168                        '''
169                        A ship "trip" in a zone consists of:
170
171                        MMSI
172                        ZONE_ENTER - zone entrance time (nanos)
173                        ZONE_EXIT - zone exit time (nanos)
174                        DESIG - ship desig
175                        CPA - ship closest point of approach to sensor (km)
176                        CPA_TIME - time to cpa (nanos)
177                        MSG_INFO - [[MSG_NANOS, LAT, LON, BRG, RNG, SOG], ...]
178
179                        Note: A ship may enter and leave a zone more than once 
180                        per day. Also, if a ships slows to < 1.0, then it falls out
181                        of the detection and buffer zone, and into the no-ship
182                        zone.
183                        '''
184                        # What zone is the ship in?
185                        if detection_zone_min <= rng <= detection_zone_max and spd >= 1.0: # ship is in the detection zone
186                            current_zone = DETECT_ZONE
187                        elif detection_zone_max < rng <= buffer_zone_max and spd >= 1.0: # ship is in the buffer zone
188                            current_zone = BUFFER_ZONE
189                        else: # ship is in the NOSHIP zone
190                            current_zone = NOSHIP_ZONE
191
192                        # this is the start of a new trip if we've never seen this MMSI or the 
193                        # previous observation of this MMIS was in a different zone.
194                        # TODO: what if the ship turn off AIS and travel to another zone and then
195                        # returns to this zone and retranmits AIS? Should we account for time
196                        # between messages when constructing trips?
197                        new_trip = not (mmsi in ship_zone) or (current_zone != ship_zone[mmsi])
198                        # update the ship's zone
199                        ship_zone[mmsi] = current_zone
200
201                        # start a new trip, or update and existing trip
202                        if ship_zone[mmsi] == DETECT_ZONE:
203                            df_trips_detect = super().add_or_update_trip(df_trips_detect, new_trip, mmsi, msg_time_nanos, ship_desig, rng, lat, lon, brg, spd)
204                        elif ship_zone[mmsi] == BUFFER_ZONE:
205                            df_trips_buffer = super().add_or_update_trip(df_trips_buffer, new_trip, mmsi, msg_time_nanos, ship_desig, rng, lat, lon, brg, spd)
206                        else: # ship is in the NOSHIP_ZONE. don't add or update any trips
207                            continue
208                            
209            h5file.close()
210
211        if len(df_trips_detect[df_trips_detect["ZONE_EXIT"] == 0]) > 0 or \
212            len(df_trips_buffer[df_trips_buffer["ZONE_EXIT"] == 0]) > 0:
213            raise Exception("Expected ZONE_EXIT times to be > 0")
214
215        return df_trips_detect, df_trips_buffer

451    def _get_harp_trips(self,
452                    user_start_nanos: int,

456                    uscg_ais_path: str,

458        uscg_sorted_file_list = sorted(os.listdir(uscg_ais_path))
459        # get all the relevant uscg ais csv files and place them in a list
460        uscg_file_nanos = [uscg_ais_to_nanos(filename) for filename in uscg_sorted_file_list]
461        # - find the indices of the start and end nanos in uscg_file_nanos. This will give us the range of csv files to work with
462        start_index, end_index = self.start_end_indices(uscg_file_nanos, user_start_nanos, user_end_nanos)
463        # select the subsection of files
464        ais_files = uscg_sorted_file_list[start_index:end_index+1]
465        df_uscg_info_data = []
466        # process all the csv files and place in a dataframe
467        for i, ais_file in enumerate(ais_files):
468            # check for start time to concat
